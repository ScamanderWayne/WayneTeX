\lsection{Marketingový mix}
\qaw{Vyjmenuj 4P}{%
\textbl{Produkt}
-- výrobek nebo služba\\
\textbl{Place}
-- distribuční politika\\
\textbl{Price}
-- cenová politika\\
\textbl{Promotion}
-- propagace, podpora prodeje
}
\qaw{Výrobek}{%
Potřebujeme \textbl{správné informace o trhu}, kam se snažíme prorazit, a tedy \textbl{zákazníkovi, kterému chceme prodávat}.
Nesmíme \texthl{přecenit} ani \texthl{nedocenit} poptávku.
Musíme si vybrat \texthl{správný moment zavedení} výrobku na trh.
Závisí na \texthl{vzhledu}, \texthl{obalu}, \texthl{kvalitě}, \texthl{značce} i \texthl{modernitě}.\\
\begin{wnumbered}
\item \textbl{fáze}
-- zavádění výrobku na trh (nízka cena, intenzivní reklama)
\item \textbl{fáze}
-- růst (cena se zvyšuje, reklama trvá)
\item \textbl{fáze}
-- zralost (nejvyšší cena, reklama pouze připomínková)
\item \textbl{fáze}
-- prodej klesá (cena se snižuje, reklama je zbytečná)
\end{wnumbered}
}
\qaw{Distribuce}{%
Cílem distribučních cest je prodávat \textbl{ve správný moment správnému zákazníkovi}.\\
Dělíme na \textbl{přímé}, kdy zákazník nakupuje přímo od výrobce, a \textbl{nepřímé}, kam vstupují mezičlánky.\\
Mezičlánky dělíme na:\\
\texthl{prostředníky} -- skupují zboží a přeprodávají dál\\
\texthl{zprostředkovatele} -- uskutečňují kontakt mezi výrobcem a zákazníkem, ale zboží není nikdy jejich majetkem, který by pouze přeprodali (agenti)\\
\texthl{podpůrné mezičlánky} -- usnadňují směnu zboží (dopravní společnosti, reklamní agentury)\\
\vspace{1pt}\\
\textbl{Základní typy distribučních cest:}\\
výrobce -- spotřebitel\\
výrobce -- maloobchod -- spotřebitel\\
výrobce -- velkoobchod -- spotřebitel\\
výrobce -- velkoobchod -- maloobchod -- spotřebitel\\
výrobce -- zprostředkovatel -- velkoobchod -- maloobchod -- spotřebitel\\
výrobce -- velkoobchod -- zprostředkovatel -- maloobchod -- spotřebitel
}
\qaw{Cena}{%
\textbl{chci se dostat na nové trhy}
-- stanovím cenu na nižší, než je tržní, tím nalákám zákazníka od konkurence\\
\textbl{prodávání za ceny jako konkurence}
-- cenu stanovím na tržní, konkuruji pouze kvalitou výrobků\\
\textbl{politika vyšších cen}
-- cenu stanovím vysoko nad tržní, tím dávám najevo, že se jedná o kvalitní zboží, a cílím pouze na menší skupinu zákazníků, co si mohou zboží dovolit\\
\vspace{1pt}\\
\textbl{metody stanovení ceny:}\\
-- podle \texthl{nákladů}; kdy propočítáme náklady na výrobu a cenu přizpůsobíme tomu\\
-- podle \texthl{orientace na konkurenci}; sledujeme tržní cenu a dle naší politiku stanovíme nižší, stejnou nebo vyšší cenu\\
-- podle \texthl{hodnoty vnímané zákazníkem}; provedeme průzkum trhu a stanovíme cenu dle toho, jakou cenu výrobku přisuzuje zákazník (praktická jistota prodeje)\\
-- podle \texthl{orientace na poptávku}; čím je cena nižší, tím větší poptávka\\
\vspace{1pt}\\
\textbl{strategie stanovení ceny:}\\
-- \texthl{strategie pronikání}; rychlé proniknutí na trh, kde podobné zboží již existuje a my přetáhneme zákazníky svojí nízkou cenou\\
-- \texthl{strategie sbírání smetany}; výrobek je odlišný od konkurence, vysokou zaváděci cenou naznačíme jeho morální hodnotu, cílíme na inovačně zaměřené zákazníky (Mercedess, Apple, ...)
}
\qaw{Propagace}{%
\textbl{Reklama}\\
-- zprostředkovávají ji reklamní agentury; soustředí se na široké publikum; zvyšuje image firmy, odlišuje výrobek od konkurence, zvyšuje poptávku\\
-- musí být: \wemph{společensky odpovědná} (bez rasové či náboženské diskriminace), \wemph{pravdivá}, \wemph{čestná} a \wemph{slušná}\\
-- mezi hlavní média patří: televize, rozhlas, tisk, billboardy, internet\\
\vspace{1pt}\\
\textbl{Osobní prodej}\\
-- založen na přímém kontaktu; \texthl{nákladný} ale \texthl{účinný}; patří sem i prodej přes známé (Avon, Eurona)\\
\vspace{1pt}\\
\textbl{Podpora prodeje}\\
-- stimuluje prodej skrz krátkodobé akce; \texthl{kupony} či \texthl{bonusy}\\
\vspace{1pt}\\
\textbl{Sponzoring}\\
-- sponzor poskytuje peněžité či jiné prostředky a jeho partner pomáhá k dosažení marketingových cílů; vyskytuje se hlavně u kulturních, sportovních a sociálních oblastech
}
