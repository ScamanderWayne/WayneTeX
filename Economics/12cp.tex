\lsection{Obchodování s cennými papíry}
\begin{multicols}{2}
\qaw{Vyjmenuj charakteristické znaky burzy}{%
{\color{antiDP}\faCaretRight}{\color{WPP}Burza je zvláštní forma trhu s následujícími znaky}{\color{antiDP}\faCaretLeft}\\
\begin{itemize}[label=\textcolor{WPP}{$\hookrightarrow$}]
\item kupující \texthl{nemusí} toto zboží \texthl{vidět}
\item zboží má \texthl{zaručenou kvalitu} a \texthl{pravost}
\item obchody se konají v \texthl{určitou dobu} a na \texthl{určitém místě}, \texthl{pravidelně}
\item obchody podléhají \texthl{přesným pravidlům} stanovených \texthl{organizátory}, stejně jako \texthl{zákony}
\item účastníky burzy je jen \texthl{omezený počet lidí} -- \emph{makléři}; \emph{burzovní dohodci}; všichni, co \texthl{mají povolení od organizátorů}
\end{itemize}
}
\columnbreak
\qaw{Popiš všeobecné druhy burz}{%
Obchoduje se zde s \texthl{více druhy zboží}, jako jsou \emph{cenné papíry} (akcie, směnky, ...), \emph{zemědělské plodiny}, \emph{drahé kovy} atd.
}
\qaw{Popiš specializované druhy burz}{%
\begin{itemize}[label={\color{WPP}\faSlackHash}]
\item \textbl{zbožové} (komoditní) -- káva, kakao, obilí, maso, ropa, drahé kovy
\item \textbl{devizové} -- obchod s devizami založené na pohybu kurzů
\item \textbl{burzu cenných papírů} -- cenné papíry kapitálového trhu (dlouhodobé), \emph{s CP peněžního trhu se neobchoduje}
\end{itemize}
}
\end{multicols}
\qaw{Co je to burza}{%
Místo, kde se setkává \texthl{Nabídka} s \texthl{Poptávkou}; vytváří \texthl{transparentní prostředí} pro obchod; společnost vstupem na burzu \texthl{získává kapitál} a \texthl{určitou prestiž}.
}
\qaw{Druhy burz v ČR}{%
\begin{itemize}
\item[ ] \textbl{BCCP Praha, a. s.} -- burza cenných papírů
\item[ ] \textbl{RM-systém} -- cenné papíry
\item[ ] \textbl{Energetická burza Praha} -- obchody velkoodběratelů s elektřinou
\end{itemize}
}
\qaw{Jak může fyzická osoba obchodovat na burze}{%
Fyzická osoba může obchodovat na burze \texthl{přes burzovního makléře}, se kterým \texthl{má uzavřenou smlouvu}; může obchodovat on-line, platí \texthl{poplatek z objemu} obchodu makléři; majitel akcie \texthl{je evidován} v \texthl{centrálním depozitáři cenných papírů}.
}
\begin{multicols}{2}
\qaw{BCPP a.s.}{%
Vznikla \texthl{roku 1993} a je akciovou společností.
Řídí se \texthl{Zákonem o burze cenných papírů} č. 214/92sb.
Má vnitřní \texthl{Burzovní pravidla} a \texthl{Burzovní řád}:\\
nejmenší obchodovatelné množství 1 lot;
lot je standardizované množství akcií (kupříkladu 5, 100, 1000);
účtuje si poplatek z objemu obchodu.
Obchoduje se zde každý den.
}
\columnbreak
\qaw{RM-systém}{%
Je akciovou společností a druhou burzou v České Republice.
Obchoduje se zde denně, ale pro menší objem obchodů.
}
\end{multicols}
\wbreak
\qaw{Co je to cenný papír}{%
Listina (či elektronický záznam), která \texthl{ztělesňuje majetková práva} (jako je právo na podíl na zisku, splacení dlužné částky, ...).
Typickými příklady jsou \texthl{akcie}, \texthl{dluhopisy} nebo \texthl{směnky}.
}
\qaw{Akcie}{%
Představuje \texthl{podíl na majetku} a. s.;
právo \texthl{na účasti} a \texthl{hlasování} na valné hromadě;
právo na odkup akcií v případě nabídky;
\texthl{podíl na likvidačním zůstatku}.
}
